187  static Matrix4x4 rotationAround(const Vector3 &axis, double angle)
188  {
189  Vector3 axisNormalized = axis.normalized();
190  double q0 = cos(angle/2);
191  double q1 = sin(angle/2) * axisNormalized.x();
192  double q2 = sin(angle/2) * axisNormalized.y();
193  double q3 = sin(angle/2) * axisNormalized.z();
194 
195  Matrix4x4 result;
196  result(0,0) = q0*q0 + q1*q1 - q2*q2 - q3*q3;
197  result(0,1) = 2*(q1*q2 - q0*q3);
198  result(0,2) = 2*(q1*q3 + q0*q2);
199  result(1,0) = 2*(q2*q1 + q0*q3);
200  result(1,1) = (q0*q0 - q1*q1 + q2*q2 - q3*q3);
201  result(1,2) = 2*(q2*q3 - q0*q1);
202  result(2,0) = 2*(q3*q1 - q0*q2);
203  result(2,1) = 2*(q3*q2 + q0*q1);
204  result(2,2) = (q0*q0 - q1*q1 - q2*q2 + q3*q3);
205  return result;
206  }
