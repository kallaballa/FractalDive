71  void lookAt(const geom3d::Point3 &target, const geom3d::Vector3 &up)
72  {
73  // Compute vector from target to eye
74  geom3d::Vector3 back = geom3d::Vector3(m_transform.position() - target).normalized();
75 
76  // Normalize and orthogonalize view up vector
77  geom3d::Vector3 right = up.normalized().cross(back);
78  geom3d::Vector3 upOrtho = back.cross(right);
79 
80  // Set rotation of transformation matrix based on view direction and view
81  // up vector.
82  geom3d::Matrix4x4 M = m_transform.matrix();
83 
84  // First COLUMN is right vector // First ROW is right vector
85  M(0,0) = right(0); // M(0,0) = right(0);
86  M(1,0) = right(1);
87  M(2,0) = right(2);
88 
89  // Second COLUMN is up vector // Second ROW is up vector
90  M(0,1) = upOrtho(0);
91  M(1,1) = upOrtho(1);
92  M(2,1) = upOrtho(2);
93 
94  // Third COLUMN is back vector
95  M(0,2) = back(0);
96  M(1,2) = back(1);
97  M(2,2) = back(2);
98
