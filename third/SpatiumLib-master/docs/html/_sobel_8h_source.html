31  template<typename T>
32  bool apply(const Image<T, 1> &input, Image<T, 1> &output)
33  {
34  if (input.width() != output.width() ||
35  input.height() != output.height())
36  {
37  return false;
38  }
39 
40  // Iterate pixels
41  for (size_t y = 1; y < input.height()-1; y++)
42  {
43  for (size_t x = 1; x < input.width()-1; x++)
44  {
45  double Gx = input.pixel(x-1, y-1)[0] * -1 +
46  input.pixel(x-1, y)[0] * -2 +
47  input.pixel(x-1, y+1)[0] * -1 +
48  input.pixel(x+1, y-1)[0] * 1 +
49  input.pixel(x+1, y)[0] * 2 +
50  input.pixel(x+1, y+1)[0] * 1;
51 
52  double Gy = input.pixel(x-1, y-1)[0] * -1 +
53  input.pixel( x, y-1)[0] * -2 +
54  input.pixel(x+1, y-1)[0] * -1 +
55  input.pixel(x-1, y+1)[0] * 1 +
56  input.pixel( x, y+1)[0] * 2 +
57  input.pixel(x+1, y+1)[0] * 1;
58 
59  T G = std::sqrt(Gx * Gx + Gy * Gy);
60  output.pixel(x, y) = { G };
61  }
62  }
63 
64  return true;
65  }
