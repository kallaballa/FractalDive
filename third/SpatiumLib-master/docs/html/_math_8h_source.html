69 inline int solveQuadratic(double a, double b, double c, double &x1, double &x2)
70 {
71  // Compute discriminant
72  double D = (b * b) - (4 * a * c);
73  if (D < 0)
74  {
75  return 0;
76  }
77 
78  // Calculate x's
79  x1 = (-b - sqrt(D)) / (2 * a);
80  x2 = (-b + sqrt(D)) / (2 * a);
81 
82  // Swap if necessary
83  if (x1 > x2)
84  {

86  }
87

89  if (D > 0)
90  {
91  return 2;
92  }
93  else // D == 0
94  {
95  return 1;

109 inline int solveCubic(double a, double b, double c, double d, double &x1, double &x2, double &x3)
110 {
111  // http://www2.trinity.unimelb.edu.au/~rbroekst/MathX/Cubic%20Formula.pdf
112  // https://en.wikibooks.org/wiki/Trigonometry/The_solution_of_cubic_equations
113 
114  // Make a = 1. Then omit d
115  if (a != 1)
116  {
117  b /= a;
118  c /= a;
119  d /= a;
120  }
121  a = b; b = c; c = d; d = 0;
122 
123  double p = b - ((a*a) / 3);
124  double q = ((2*(a*a*a)) / 27) - ((a*b) / 3) + c;
125  //double D = ((q*q) / 4) + ((p*p*p) / 27);
126 
127  double Q = (3*b - a*a) / 9;
128  double R = (9*a*b - 27*c - 2*a*a*a) / 54;
129  double D = Q*Q*Q + R*R;
130 
131  if (D < std::numeric_limits<double>::epsilon()
132  && D > -std::numeric_limits<double>::epsilon()) // D == 0
133  {
134  // There are TWO real solutions
135  x1 = -2*cbrt(q/2) - (a/3);
136  x2 = cbrt(q/2) - (a/3);
137 
138  // Reorder results
139  if (x1 > x2)
140  {
141  std::swap(x1, x2);
142  }
143  x3 = x2;
144 
145  return 2;
146  }
147  else if (D > 0)
148  {
149  // There is ONE real solution
150  x1 = x2 = x3 = cbrt(-1*(q/2) + sqrt(D)) + cbrt(-1*(q/2) - sqrt(D)) - (a/3);
151 
152  return 1;
153  }
154  else if(D < 0)
155  {
156  // There are THREE real solution
157 
158  // Solve with trigonometry
159  double theta = acos(R/sqrt(-(Q*Q*Q)));
160  x1 = 2 * sqrt(-Q) * cos(theta / 3) - (a/3);
161  x2 = 2 * sqrt(-Q) * cos(theta / 3 + (2*PI/3)) - (a/3);
162  x3 = 2 * sqrt(-Q) * cos(theta / 3 + (4*PI/3)) - (a/3);
163 
164  // Reorder results
165  if (x1 > x2)
166  {
167  std::swap(x1, x2);
168  }
169  if (x1 > x3)
170  {
171  std::swap(x1, x3);
172  }
173  if (x2 > x3)
174  {
175  std::swap(x2, x3);
176  }
177 
178  return 3;
179  }
180 }
