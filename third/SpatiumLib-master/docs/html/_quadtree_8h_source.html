83  static PointQuadtree buildFromPoints(const std::vector<std::array<double, 2>> &points, size_t maxPointCountLeaf = 100)
84  {
85  if (points.size() == 0)
86  {
87  BoundingBox<double,2> b;
88  PointQuadtree p(b);
89  return p;
90  }
91 
92  // Compute minimum bounding rectangle
93  BoundingBox<double,2> bounds = BoundingBox<double,2>::fromPoints(points);
94 
95  // Derive minimum bounding square
96  double width = bounds.diameter(0);
97  double height = bounds.diameter(1);
98  std::array<double, 2> min = bounds.min();
99  std::array<double, 2> max = bounds.max();
100  if (width > height) {
101  max = { min[0] + width, min[1] + width };
102  } else {
103  max = { min[0] + height, min[1] + height };
104  }
105  bounds = BoundingBox<double,2>::fromMinMax(min, max);
106 
107  // Create inital tree
108  PointQuadtree tree(bounds, maxPointCountLeaf);
109 
110  // Add points
111  for(auto point : points)
112  {
113  tree.addPoint(point);
114  }
115 
116  return tree;
117  }

128  bool addPoint(const std::array<double, 2> &point)
129  {
130  if (!m_boundingBox.isInside(point))
131  {
132  return false;
133  }
134 
135  //std::shared_ptr<TreeNode<std::vector<std::array<double,2>>>> n = this->m_root;
136 
137  std::shared_ptr<PointQuadtreeNode> node = std::static_pointer_cast<PointQuadtreeNode>(this->m_root);
138  BoundingBox<double,2> bounds = m_boundingBox;
139  while(node != nullptr)
140  {
141  // Check: Is leaf node?
142  if (node->hasChildren())
143  {
144  // No leaf node. Pass to one of children.
145  // Determine child and traverse into
146  size_t childIndex = PointQuadtree::determineChild(bounds, point, bounds);
147  node = std::static_pointer_cast<PointQuadtreeNode>(node->child(childIndex));
148  }
149  else
150  {
151  // Leaf node. Add point. May need splitting child.
152  if (node->object().size() + 1 > m_maxPointCountLeaf)
153  {
154  // Split node. Create 4 children.
155  node->createChildren(node);
156 
157  // Distribute points of current node over 4 new children
158  for (size_t i = 0; i < node->object().size(); i++)
159  {
160  std::array<double, 2> childPoint = node->object()[i];
161 
162  BoundingBox<double,2> curBounds;
163  size_t curChildIndex = PointQuadtree::determineChild(bounds, childPoint, curBounds);
164  std::shared_ptr<PointQuadtreeNode> curChild = std::static_pointer_cast<PointQuadtreeNode>(node->child(curChildIndex));
165  curChild->object().push_back(childPoint);
166  }
167  node->object().clear();
168 
169  // Determine next child and traverse into
170  size_t childIndex = PointQuadtree::determineChild(bounds, point, bounds);
171  node = std::static_pointer_cast<PointQuadtreeNode>(node->child(childIndex));
172  }
173  else
174  {
175  // Simply add point
176  node->object().push_back(point);
177 
178  // Stop tree traversal
179  node = nullptr;
180  }
181  }
182  }
183 
184  return true;
185  }

194  static size_t determineChild(const BoundingBox<double,2> &boundsParent, const std::array<double, 2> &point, BoundingBox<double,2> &boundsChild)
195  {
196  std::array<double, 2> min = boundsParent.min();
197  std::array<double, 2> max = boundsParent.max();
198  std::array<double, 2> center = boundsParent.center();
199  if (point[0] < center[0])
200  {
201  // Left half
202  if (point[1] < center[1])
203  {
204  // Bottom
205  boundsChild = BoundingBox<double,2>::fromMinMax(min, center);
206  return 0;
207  }
208  else
209  {
210  // Top
211  boundsChild = BoundingBox<double,2>::fromMinMax({ min[0], center[1] }, { center[0], max[1] });
212  return 2;
213  }
214  }
215  else
216  {
217  // Right half
218  if (point[1] < center[1])
219  {
220  // Top
221  boundsChild = BoundingBox<double,2>::fromMinMax({ center[0], min[1] }, { max[0], center[1] });
222  return 1;
223  }
224  else
225  {
226  // Bottom
227  boundsChild = BoundingBox<double,2>::fromMinMax(center, max);
228  return 3;
229  }
230  }
231  }
