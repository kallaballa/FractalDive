51  bool apply(const Image<T, 3> &input, Image<T, 1> &output)
52  {
53  // Check image sizes

55  input.height() != output.height())
56  {
57  return false;

59

61  {
62  for (size_t x = 0; x < input.width(); x++)
63  {
64  std::array<T, 3> pixel = input.pixel(x, y);
65  output.pixel(x, y) = { pixel[0] * m_redCoeff + pixel[1] * m_greenCoeff + pixel[2] * m_blueCoeff };
