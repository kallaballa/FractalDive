72  static BoundingBox<T,D> fromPoints(const std::vector<std::array<T,D>> &points)
73  {
74  if (points.size() == 0)
75  {
76  return BoundingBox();
77  }
78 
79  // Set initial bounds
80  std::array<T, D> minVal, maxVal;
81  for (size_t i = 0; i < D; i++)
82  {
83  minVal[i] = maxVal[i] = points[0][i];
84  }
85 
86  // Iterate points and update bounds
87  for (size_t i = 1; i < points.size(); i++)
88  {
89  std::array<T, D> point = points[i];
90 
91  // Iterate dimensions
92  for (size_t i = 0; i < D; i++)
93  {
94  if (point[i] < minVal[i]) {
95  minVal[i] = point[i];
96  } else if(point[i] > maxVal[i]) {
97  maxVal[i] = point[i];
98  }
99  }
100  }
101 
102  return BoundingBox<T,D>::fromMinMax(minVal, maxVal);
103  }

105  static BoundingBox<T,D> fromMinMax(const std::array<T,D> &min, const std::array<T,D> &max)
106  {
107  // Compute center and radii
108  std::array<T,D> centerVal, radiiVal;
109  for (size_t i = 0; i < D; i++)
110  {
111  centerVal[i] = max[i] * 0.5 + min[i] * 0.5;
112  radiiVal[i] = max[i] * 0.5 - min[i] * 0.5;
113  }
114
